    ```rust
    Commands::Audit { r#type, since, detail, limit } => {
        // 1. Parse --since into epoch millis: "1h" -> now - 3600_000, "24h" -> now - 86400_000, "7d" -> now - 604800_000
        let since_ts = since.as_deref().map(parse_duration_ago);

        // 2. Connect to daemon and send AuditQuery
        let entries = client::send_audit_query(
            r#type.clone(),
            since_ts,
            Some(limit),
        ).await?;

        // 3. If --detail specified, find entry by ID and print full detail
        if let Some(detail_id) = &detail {
            if let Some(entry) = entries.iter().find(|e| e.id == *detail_id) {
                print_audit_detail(entry);
            } else {
                eprintln!("No audit entry found with ID: {}", detail_id);
            }
            return Ok(());
        }

        // 4. Print table of entries
        if entries.is_empty() {
            println!("No audit entries found.");
            return Ok(());
        }
        for entry in &entries {
            print_audit_row(entry);
        }
        println!("\n{} entries shown. Use --detail <id> for full trace.", entries.len());
    }
    ```

    ```rust
    use amux_protocol::messages::AuditEntryPublic;

    pub async fn send_audit_query(
        action_types: Option<Vec<String>>,
        since: Option<u64>,
        limit: Option<usize>,
    ) -> Result<Vec<AuditEntryPublic>> {
        let mut client = connect().await?;
        client.send(ClientMessage::AuditQuery { action_types, since, limit }).await?;
        match client.receive().await? {
            DaemonMessage::AuditList { entries_json } => {
                let entries: Vec<AuditEntryPublic> = serde_json::from_str(&entries_json)?;
                Ok(entries)
            }
            other => anyhow::bail!("Unexpected response: {:?}", other),
        }
    }
    ```

    ```rust
    /// Cancel active escalation and return control to user. Per D-13.
    /// Returns Ok(message) on success, Err on failure (e.g., no active escalation).
    pub fn cancel_escalation(state: &mut EscalationState) -> Result<String> {
        if state.current_level == EscalationLevel::SelfCorrection && state.total_escalations == 0 {
            anyhow::bail!("No active escalation to cancel");
        }
        let prev_level = state.current_level.clone();
        state.escalation_history.push(EscalationEvent {
            level: prev_level.clone(),
            reason: "User cancelled escalation (I'll handle this)".to_string(),
            timestamp: std::time::SystemTime::now()
                .duration_since(std::time::UNIX_EPOCH)
                .unwrap()
                .as_millis() as u64,
            outcome: Some("cancelled_by_user".to_string()),
        });
        // Reset to L0 (self-correction) -- user takes over
        state.current_level = EscalationLevel::SelfCorrection;
        state.attempts_at_level = 0;
        Ok(format!("Escalation cancelled at {:?}. You now have control.", prev_level))
    }
    ```

    3. Broadcast `AgentEvent::EscalationUpdate` with from_level={prev}, to_level="L0", reason="User took over"
    4. Broadcast `AgentEvent::AuditAction` with the cancellation entry
    5. Handle race condition per Pitfall 6: if escalation already resolved, return success with "Escalation already resolved"
