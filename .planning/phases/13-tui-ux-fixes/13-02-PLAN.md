    3. Add `render_features_tab()` function:

    ```rust
    fn render_features_tab<'a>(
        settings: &'a SettingsState,
        config: &'a ConfigState,
        tier: &crate::state::tier::TierState,
        theme: &ThemeTokens,
    ) -> Vec<Line<'a>> {
        let mut lines = Vec::new();

        // Read values from agent_config_raw
        let raw = config.agent_config_raw.as_ref();

        // Section: Tier & Security
        lines.push(Line::raw(""));
        lines.push(Line::from(Span::styled("  Tier & Security", theme.fg_active)));
        lines.push(Line::raw(""));

        // Field 0: tier_override (cycle: newcomer / familiar / power_user / expert)
        let tier_val = raw
            .and_then(|r| r.get("tier"))
            .and_then(|t| t.get("user_override"))
            .and_then(|v| v.as_str())
            .unwrap_or(&tier.current_tier);
        render_field_line(&mut lines, settings, 0, "Tier Override", tier_val, theme);

        // Field 1: managed_security_level (cycle: permissive / balanced / strict)
        let security_val = raw
            .and_then(|r| r.get("managed_security_level"))
            .and_then(|v| v.as_str())
            .unwrap_or("balanced");
        render_field_line(&mut lines, settings, 1, "Security Level", security_val, theme);

        // Section: Heartbeat
        lines.push(Line::raw(""));
        lines.push(Line::from(Span::styled("  Heartbeat", theme.fg_active)));
        lines.push(Line::raw(""));

        // Field 2: heartbeat.cron (editable text, e.g. "*/15 * * * *")
        let cron_val = raw
            .and_then(|r| r.get("heartbeat"))
            .and_then(|h| h.get("cron"))
            .and_then(|v| v.as_str())
            .unwrap_or("*/15 * * * *");
        render_field_line(&mut lines, settings, 2, "Cron Schedule", cron_val, theme);

        // Field 3: heartbeat.quiet_start (editable, e.g. "22:00")
        let quiet_start = raw
            .and_then(|r| r.get("heartbeat"))
            .and_then(|h| h.get("quiet_start"))
            .and_then(|v| v.as_str())
            .unwrap_or("22:00");
        render_field_line(&mut lines, settings, 3, "Quiet Start", quiet_start, theme);

        // Field 4: heartbeat.quiet_end (editable, e.g. "07:00")
        let quiet_end = raw
            .and_then(|r| r.get("heartbeat"))
            .and_then(|h| h.get("quiet_end"))
            .and_then(|v| v.as_str())
            .unwrap_or("07:00");
        render_field_line(&mut lines, settings, 4, "Quiet End", quiet_end, theme);

        // Field 5-8: heartbeat check toggles (stale_todos, stuck_goals, unreplied_msgs, repo_changes)
        for (idx, (key, label)) in [
            ("check_stale_todos", "Check Stale Todos"),
            ("check_stuck_goals", "Check Stuck Goals"),
            ("check_unreplied_messages", "Check Unreplied Msgs"),
            ("check_repo_changes", "Check Repo Changes"),
        ].iter().enumerate() {
            let enabled = raw
                .and_then(|r| r.get("heartbeat"))
                .and_then(|h| h.get(key))
                .and_then(|v| v.as_bool())
                .unwrap_or(true);
            render_toggle_line(&mut lines, settings, 5 + idx, label, enabled, theme);
        }

        // Section: Memory & Learning
        lines.push(Line::raw(""));
        lines.push(Line::from(Span::styled("  Memory & Learning", theme.fg_active)));
        lines.push(Line::raw(""));

        // Field 9: consolidation.enabled (toggle)
        let consol_enabled = raw
            .and_then(|r| r.get("consolidation"))
            .and_then(|c| c.get("enabled"))
            .and_then(|v| v.as_bool())
            .unwrap_or(true);
        render_toggle_line(&mut lines, settings, 9, "Consolidation", consol_enabled, theme);

        // Field 10: consolidation.decay_half_life_hours (editable number)
        let decay_val = raw
            .and_then(|r| r.get("consolidation"))
            .and_then(|c| c.get("decay_half_life_hours"))
            .and_then(|v| v.as_f64())
            .map(|v| format!("{:.0}", v))
            .unwrap_or_else(|| "69".to_string());
        render_field_line(&mut lines, settings, 10, "Decay Half-Life (hrs)", &decay_val, theme);

        // Field 11: heuristic_promotion_threshold (editable number)
        let heur_val = raw
            .and_then(|r| r.get("consolidation"))
            .and_then(|c| c.get("heuristic_promotion_threshold"))
            .and_then(|v| v.as_u64())
            .map(|v| v.to_string())
            .unwrap_or_else(|| "5".to_string());
        render_field_line(&mut lines, settings, 11, "Heuristic Threshold", &heur_val, theme);

        // Section: Skills
        lines.push(Line::raw(""));
        lines.push(Line::from(Span::styled("  Skills", theme.fg_active)));
        lines.push(Line::raw(""));

        // Field 12: skill_discovery.enabled (toggle)
        let skill_enabled = raw
            .and_then(|r| r.get("skill_discovery"))
            .and_then(|s| s.get("enabled"))
            .and_then(|v| v.as_bool())
            .unwrap_or(true);
        render_toggle_line(&mut lines, settings, 12, "Auto-Discovery", skill_enabled, theme);

        // Field 13: skill_discovery.promotion_threshold (editable number)
        let promo_val = raw
            .and_then(|r| r.get("skill_discovery"))
            .and_then(|s| s.get("promotion_threshold"))
            .and_then(|v| v.as_u64())
            .map(|v| v.to_string())
            .unwrap_or_else(|| "3".to_string());
        render_field_line(&mut lines, settings, 13, "Promotion Threshold", &promo_val, theme);

        lines
    }
    ```

    Create two helper functions `render_field_line` and `render_toggle_line` that follow the same cursor/marker/style pattern used in `render_advanced_tab` (field_cursor check, "> " marker when selected, edit buffer display when editing, etc.). These helpers will reduce code duplication.
