```yaml
# tamux CI/CD Pipeline
# Triggered by vX.Y.Z tags for release builds
# Produces: platform binaries, Electron desktop apps, GitLab Release, npm publish

stages:
  - build
  - package
  - release

variables:
  CARGO_HOME: "${CI_PROJECT_DIR}/.cargo"
  RUSTUP_HOME: "${CI_PROJECT_DIR}/.rustup"

# Cache Rust compilation across jobs
.rust-cache:
  cache:
    key: "${CI_COMMIT_REF_SLUG}-rust"
    paths:
      - .cargo/registry/
      - target/

# ------------------------------------------------------------------
# Stage: build (Rust binaries per platform)
# ------------------------------------------------------------------

.rust-build:
  stage: build
  extends: .rust-cache
  script:
    - export VERSION="${CI_COMMIT_TAG#v}"
    - ./scripts/build-release.sh --skip-frontend --skip-electron --target "$RUST_TARGET"
    # Collect binaries into a tarball per D-08 naming convention
    - mkdir -p artifacts
    - |
      if [ "$PLATFORM" = "windows-x64" ]; then
        BINS="tamux-daemon.exe tamux.exe tamux-tui.exe"
      else
        BINS="tamux-daemon tamux tamux-tui"
      fi
    - tar czf "artifacts/tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz" -C "dist-release/${DIST_DIR}" $BINS
    # Generate per-platform SHA256SUMS
    - cd artifacts
    - sha256sum "tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz" > "SHA256SUMS-${PLATFORM}.txt"
    # Upload to GitLab Generic Package Registry
    - |
      for FILE in tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz SHA256SUMS-${PLATFORM}.txt; do
        curl --fail --header "JOB-TOKEN: ${CI_JOB_TOKEN}" \
          --upload-file "$FILE" \
          "${CI_API_V4_URL}/projects/${CI_PROJECT_ID}/packages/generic/tamux/${VERSION}/${FILE}"
      done
  artifacts:
    paths:
      - artifacts/
    expire_in: 1 day
  rules:
    - if: '$CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/'

build:linux-x64:
  extends: .rust-build
  image: rust:latest
  tags: [linux, x64]
  variables:
    RUST_TARGET: x86_64-unknown-linux-gnu
    PLATFORM: linux-x64
    DIST_DIR: linux

build:linux-arm64:
  extends: .rust-build
  image: rust:latest
  tags: [linux, x64]
  variables:
    RUST_TARGET: aarch64-unknown-linux-gnu
    PLATFORM: linux-arm64
    DIST_DIR: linux
  before_script:
    - cargo install cross --locked 2>/dev/null || true
    - rustup target add aarch64-unknown-linux-gnu 2>/dev/null || true
  script:
    - export VERSION="${CI_COMMIT_TAG#v}"
    # Use cross for ARM64 cross-compilation (Pitfall 3)
    - cross build --release --target aarch64-unknown-linux-gnu
    - mkdir -p "dist-release/linux"
    - cp target/aarch64-unknown-linux-gnu/release/tamux-daemon dist-release/linux/
    - cp target/aarch64-unknown-linux-gnu/release/tamux dist-release/linux/
    - cp target/aarch64-unknown-linux-gnu/release/tamux-tui dist-release/linux/
    - mkdir -p artifacts
    - tar czf "artifacts/tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz" -C dist-release/linux tamux-daemon tamux tamux-tui
    - cd artifacts
    - sha256sum "tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz" > "SHA256SUMS-${PLATFORM}.txt"
    - |
      for FILE in tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz SHA256SUMS-${PLATFORM}.txt; do
        curl --fail --header "JOB-TOKEN: ${CI_JOB_TOKEN}" \
          --upload-file "$FILE" \
          "${CI_API_V4_URL}/projects/${CI_PROJECT_ID}/packages/generic/tamux/${VERSION}/${FILE}"
      done

build:darwin-arm64:
  extends: .rust-build
  tags: [macos, arm64]
  variables:
    RUST_TARGET: aarch64-apple-darwin
    PLATFORM: darwin-arm64
    DIST_DIR: macos
  rules:
    - if: '$CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/'
      when: manual
      allow_failure: true

build:windows-x64:
  extends: .rust-build
  image: rust:latest
  tags: [linux, x64]
  variables:
    RUST_TARGET: x86_64-pc-windows-gnu
    PLATFORM: windows-x64
    DIST_DIR: windows
  before_script:
    - apt-get update -qq && apt-get install -y -qq gcc-mingw-w64-x86-64 2>/dev/null || true
    - rustup target add x86_64-pc-windows-gnu

# ------------------------------------------------------------------
# Stage: package (Electron desktop apps)
# ------------------------------------------------------------------

.electron-package:
  stage: package
  image: node:22
  script:
    - export VERSION="${CI_COMMIT_TAG#v}"
    - cd frontend
    - npm ci
    - npm run build
    # Copy Rust binaries from build stage into dist/ for electron-builder extraResources
    - mkdir -p dist
    - cp ../artifacts/tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz /tmp/
    - cd /tmp && tar xzf tamux-binaries-v${VERSION}-${PLATFORM}.tar.gz && cd -
    - cp /tmp/tamux-daemon${BIN_EXT} dist/ 2>/dev/null || true
    - cp /tmp/tamux${BIN_EXT} dist/ 2>/dev/null || true
    # Build Electron app
    - npx electron-builder --${EB_PLATFORM} --publish never
    # Upload desktop artifacts to package registry
    - |
      for FILE in ${DESKTOP_ARTIFACTS}; do
        if [ -f "$FILE" ]; then
          BASENAME=$(basename "$FILE")
          curl --fail --header "JOB-TOKEN: ${CI_JOB_TOKEN}" \
            --upload-file "$FILE" \
            "${CI_API_V4_URL}/projects/${CI_PROJECT_ID}/packages/generic/tamux/${VERSION}/tamux-desktop-v${VERSION}-${DESKTOP_NAME}"
        fi
      done
  rules:
    - if: '$CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/'
      when: manual
      allow_failure: true

package:linux-x64:
  extends: .electron-package
  tags: [linux, x64]
  needs: ["build:linux-x64"]
  variables:
    PLATFORM: linux-x64
    BIN_EXT: ""
    EB_PLATFORM: linux
    DESKTOP_NAME: "linux-x64.AppImage"
    DESKTOP_ARTIFACTS: "release/*.AppImage"

package:windows-x64:
  extends: .electron-package
  tags: [linux, x64]
  needs: ["build:windows-x64"]
  variables:
    PLATFORM: windows-x64
    BIN_EXT: ".exe"
    EB_PLATFORM: win
    DESKTOP_NAME: "windows-x64.exe"
    DESKTOP_ARTIFACTS: "release/*.exe"

# ------------------------------------------------------------------
# Stage: release (GitLab Release + npm publish)
# ------------------------------------------------------------------

release:create:
  stage: release
  image: registry.gitlab.com/gitlab-org/cli:latest
  needs:
    - job: build:linux-x64
    - job: build:windows-x64
    - job: build:linux-arm64
      optional: true
    - job: build:darwin-arm64
      optional: true
  script:
    - export VERSION="${CI_COMMIT_TAG#v}"
    - export PKG_BASE="${CI_API_V4_URL}/projects/${CI_PROJECT_ID}/packages/generic/tamux/${VERSION}"
    # Generate combined SHA256SUMS
    - |
      echo "# SHA256SUMS for tamux v${VERSION}" > SHA256SUMS.txt
      for PLATFORM in linux-x64 linux-arm64 darwin-arm64 windows-x64; do
        SUMS_FILE="SHA256SUMS-${PLATFORM}.txt"
        curl -fsSL --header "JOB-TOKEN: ${CI_JOB_TOKEN}" \
          "${PKG_BASE}/${SUMS_FILE}" >> SHA256SUMS.txt 2>/dev/null || true
      done
    - |
      curl --fail --header "JOB-TOKEN: ${CI_JOB_TOKEN}" \
        --upload-file SHA256SUMS.txt \
        "${PKG_BASE}/SHA256SUMS.txt"
    # Create GitLab Release with asset links
    - |
      glab release create "${CI_COMMIT_TAG}" \
        --name "tamux v${VERSION}" \
        --notes "## tamux v${VERSION}

      ### Binary Downloads
      | Platform | Download |
      |----------|----------|
      | Linux x64 | [tamux-binaries-v${VERSION}-linux-x64.tar.gz](${PKG_BASE}/tamux-binaries-v${VERSION}-linux-x64.tar.gz) |
      | Linux ARM64 | [tamux-binaries-v${VERSION}-linux-arm64.tar.gz](${PKG_BASE}/tamux-binaries-v${VERSION}-linux-arm64.tar.gz) |
      | macOS ARM64 | [tamux-binaries-v${VERSION}-darwin-arm64.tar.gz](${PKG_BASE}/tamux-binaries-v${VERSION}-darwin-arm64.tar.gz) |
      | Windows x64 | [tamux-binaries-v${VERSION}-windows-x64.tar.gz](${PKG_BASE}/tamux-binaries-v${VERSION}-windows-x64.tar.gz) |

      ### Checksums
      [SHA256SUMS.txt](${PKG_BASE}/SHA256SUMS.txt)

      ### Install
      \`\`\`
      npx tamux
      # or
      curl -fsSL https://tamux.dev/install.sh | sh
      \`\`\`
      "
  rules:
    - if: '$CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/'

release:npm-publish:
  stage: release
  image: node:22
  needs: ["release:create"]
  script:
    - cd npm-package
    - echo "//registry.npmjs.org/:_authToken=${NPM_TOKEN}" > .npmrc
    - npm publish --access public
  rules:
    - if: '$CI_COMMIT_TAG =~ /^v\d+\.\d+\.\d+$/'
      when: manual
      allow_failure: true
```

Key design decisions:
- linux-x64 and windows-x64 builds are automatic; darwin-arm64 is manual (requires macOS runner -- Pitfall 2)
- linux-arm64 uses `cross` for cross-compilation from x64 (Pitfall 3)
- Package registry URLs use `${CI_API_V4_URL}/projects/${CI_PROJECT_ID}` format (Pitfall 4)
- Electron packaging is manual (separate stage, allow_failure: true) per D-07 split artifacts
- npm publish is manual with NPM_TOKEN CI variable requirement per D-09
- Combined SHA256SUMS.txt aggregated in release stage
- glab CLI (not deprecated release-cli) for release creation
