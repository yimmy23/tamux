    **Frontend: statusStore.ts** — create `frontend/src/lib/statusStore.ts`:

    ```typescript
    import { create } from "zustand";
    import { getBridge } from "./bridge";

    export type AgentActivityState =
        | "idle"
        | "thinking"
        | "executing_tool"
        | "waiting_for_approval"
        | "running_goal";

    interface ProviderHealth {
        provider: string;
        status: string;
        failures: number;
    }

    interface GatewayStatus {
        platform: string;
        status: string;
        lastCheck: number;
    }

    interface RecentAction {
        actionType: string;
        explanation: string;
        timestamp: number;
    }

    interface StatusState {
        activity: AgentActivityState;
        activeThreadId: string | null;
        activeGoalRunId: string | null;
        activeGoalRunTitle: string | null;
        providerHealth: ProviderHealth[];
        gatewayStatuses: GatewayStatus[];
        recentActions: RecentAction[];
        lastUpdated: number;
        updateStatus: (data: Partial<StatusState>) => void;
    }

    export const useStatusStore = create<StatusState>((set) => ({
        activity: "idle",
        activeThreadId: null,
        activeGoalRunId: null,
        activeGoalRunTitle: null,
        providerHealth: [],
        gatewayStatuses: [],
        recentActions: [],
        lastUpdated: 0,
        updateStatus: (data) => set({ ...data, lastUpdated: Date.now() }),
    }));

    let _pollInterval: ReturnType<typeof setInterval> | null = null;

    /** Start polling daemon for status updates every 10 seconds. */
    export function hydrateStatusStore(): void {
        if (_pollInterval) return;
        pollStatus();
        _pollInterval = setInterval(pollStatus, 10_000);
    }

    async function pollStatus(): Promise<void> {
        const bridge = getBridge();
        if (!bridge) return;
        try {
            // Use the dedicated agentGetStatus IPC call which sends AgentStatusQuery to daemon
            const status = await bridge.agentGetStatus?.();
            if (!status) return;

            const { updateStatus } = useStatusStore.getState();

            // Parse activity state
            const validActivities: AgentActivityState[] = [
                "idle", "thinking", "executing_tool", "waiting_for_approval", "running_goal"
            ];
            const activity = validActivities.includes(status.activity as AgentActivityState)
                ? (status.activity as AgentActivityState)
                : "idle";

            // Parse JSON fields
            let providerHealth: ProviderHealth[] = [];
            let gatewayStatuses: GatewayStatus[] = [];
            let recentActions: RecentAction[] = [];

            try { providerHealth = JSON.parse(status.providerHealthJson || "[]"); } catch { /* empty */ }
            try { gatewayStatuses = JSON.parse(status.gatewayStatusesJson || "[]"); } catch { /* empty */ }
            try { recentActions = JSON.parse(status.recentActionsJson || "[]"); } catch { /* empty */ }

            updateStatus({
                activity,
                activeThreadId: status.activeThreadId,
                activeGoalRunId: status.activeGoalRunId,
                activeGoalRunTitle: status.activeGoalRunTitle,
                providerHealth,
                gatewayStatuses,
                recentActions,
            });
        } catch (e) {
            console.warn("[status] poll failed:", e);
        }
    }

    /** Clean up polling interval. */
    export function destroyStatusStore(): void {
        if (_pollInterval) {
            clearInterval(_pollInterval);
            _pollInterval = null;
        }
    }
    ```
