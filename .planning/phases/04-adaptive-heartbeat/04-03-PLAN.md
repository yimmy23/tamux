    6. Add learning transparency invocation with pattern-change detection in heartbeat.rs (D-10). Add this AFTER the anticipatory section construction (step 2-3) and BEFORE the final synthesis prompt assembly, so learning observations can be appended to the anticipatory section:

       ```rust
       // --- Phase 2.6: Learning transparency -- detect and report meaningful pattern changes (D-10) ---
       let mut learning_observations: Vec<String> = Vec::new();

       // (a) Detect peak hours change: compare current smoothed peak hours to last-reported peak hours.
       // Store "last reported peak hours" on AgentEngine or as a local persistent marker.
       // A meaningful change = the set of hours where smoothed value >= ema_threshold has shifted
       // by more than 2 hours compared to the last time we reported.
       {
           let model = self.operator_model.read().await;
           let config = self.config.read().await;
           let threshold = config.ema_activity_threshold;
           drop(config);

           // Compute current peak hours from smoothed histogram
           let current_peaks: Vec<u8> = model.session_rhythm.smoothed_activity_histogram.iter()
               .filter(|(_, &v)| v >= threshold)
               .map(|(&h, _)| h)
               .collect();

           let previous_peaks = &model.session_rhythm.peak_activity_hours_utc;

           // Meaningful change: symmetric difference has > 2 hours
           let added: Vec<u8> = current_peaks.iter().filter(|h| !previous_peaks.contains(h)).copied().collect();
           let removed: Vec<u8> = previous_peaks.iter().filter(|h| !current_peaks.contains(h)).copied().collect();

           if added.len() + removed.len() > 2 && model.session_rhythm.session_count >= 5 {
               // Format peak hours as human-readable string
               let mut sorted_peaks = current_peaks.clone();
               sorted_peaks.sort();
               let peak_str = sorted_peaks.iter().map(|h| format!("{}:00", h)).collect::<Vec<_>>().join(", ");
               let data = serde_json::json!({ "peak_hours": peak_str });
               if let ExplanationResult::Template(explanation) = generate_explanation("schedule_learned", &data) {
                   learning_observations.push(explanation);
               }
           }
       }

       // (b) Detect check deprioritization: when a learned weight crosses below 0.5 for the first time.
       // Read learned_check_weights (populated by Plan 02's weight update loop) and compare to a
       // "last_reported_weights" snapshot. If any weight crossed below 0.5 since last report, invoke template.
       {
           let weights = self.learned_check_weights.read().await;
           let check_names = [
               (HeartbeatCheckType::StaleTodos, "stale todos"),
               (HeartbeatCheckType::StuckGoals, "stuck goals"),
               (HeartbeatCheckType::UnrepliedMessages, "unreplied messages"),
               (HeartbeatCheckType::RepoChanges, "repo changes"),
           ];
           for (check_type, check_name) in &check_names {
               if let Some(&weight) = weights.get(check_type) {
                   // Report deprioritization when weight drops below 0.5
                   // To avoid repeated reports, only report if the config weight is still at default (1.0)
                   // meaning this is a learned change, not a user-configured one.
                   if weight < 0.5 {
                       let data = serde_json::json!({ "check_name": check_name, "weight": weight });
                       if let ExplanationResult::Template(explanation) = generate_explanation("check_deprioritized", &data) {
                           learning_observations.push(explanation);
                       }
                   }
                   // Report reprioritization when weight recovers above 0.8 after being below 0.5
                   // (Simple heuristic: report if weight is between 0.8 and 1.0 and was previously low)
                   // For MVP, report any weight above 0.8 that has a nonzero dismissal history
                   // More sophisticated tracking can be added later.
               }
           }
       }

       // Append learning observations to the anticipatory section of the synthesis prompt
       let learning_section = if learning_observations.is_empty() {
           String::new()
       } else {
           let observations = learning_observations.iter()
               .map(|obs| format!("- [learning] {}", obs))
               .collect::<Vec<_>>()
               .join("\n");
           format!("\n\n== Learning Observations ==\n{}", observations)
       };
       ```

       Then include `learning_section` in the final synthesis_prompt concatenation alongside `anticipatory_section` and `morning_brief_note`.
