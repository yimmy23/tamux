```rust
pub async fn send_config_get() -> Result<serde_json::Value> {
```

**3. Add match arm for Commands::Settings in the main match block:**

For `SettingsAction::List`:
- Call `client::send_config_get()` to get full config JSON.
- Flatten the JSON object into dot-notation key=value pairs and print each one.
- Use a recursive helper to flatten nested objects. Arrays print as JSON.
- Skip sensitive keys: any key containing "api_key", "token", "secret" should show the value as "***" (redacted).

```rust
Commands::Settings { action } => match action {
    SettingsAction::List => {
        let config = client::send_config_get().await?;
        let mut pairs: Vec<(String, String)> = Vec::new();
        flatten_json("", &config, &mut pairs);
        if pairs.is_empty() {
            println!("No configuration found.");
        } else {
            for (key, value) in &pairs {
                println!("{} = {}", key, value);
            }
        }
    }
    SettingsAction::Get { key } => {
        let config = client::send_config_get().await?;
        // Navigate dot-notation path through JSON
        let value = resolve_dot_path(&config, &key);
        match value {
            Some(v) => {
                if is_sensitive_key(&key) {
                    println!("***");
                } else {
                    // Print bare string for string values, JSON for others
                    match v.as_str() {
                        Some(s) => println!("{}", s),
                        None => println!("{}", serde_json::to_string_pretty(v).unwrap_or_default()),
                    }
                }
            }
            None => {
                eprintln!("Key not found: {}", key);
                std::process::exit(1);
            }
        }
    }
    SettingsAction::Set { key, value } => {
        // Convert dot-notation key to JSON pointer: "heartbeat.interval" -> "/heartbeat/interval"
        let json_pointer = format!("/{}", key.replace('.', "/"));
        // Try to parse value as JSON first, fall back to treating it as a JSON string
        let value_json = if serde_json::from_str::<serde_json::Value>(&value).is_ok() {
            value.clone()
        } else {
            serde_json::to_string(&value).unwrap_or_else(|_| format!("\"{}\"", value))
        };
        client::send_config_set(json_pointer, value_json).await?;
        println!("{} = {}", key, value);
    }
},
```

```rust
/// Flatten a JSON value into dot-notation key=value pairs.
fn flatten_json(prefix: &str, value: &serde_json::Value, out: &mut Vec<(String, String)>) {
    match value {
        serde_json::Value::Object(map) => {
            for (k, v) in map {
                let full_key = if prefix.is_empty() {
                    k.clone()
                } else {
                    format!("{}.{}", prefix, k)
                };
                flatten_json(&full_key, v, out);
            }
        }
        serde_json::Value::Null => {
            // Skip null values
        }
        serde_json::Value::String(s) => {
            let display = if is_sensitive_key(prefix) { "***".to_string() } else { s.clone() };
            out.push((prefix.to_string(), display));
        }
        other => {
            let display = if is_sensitive_key(prefix) {
                "***".to_string()
            } else {
                other.to_string()
            };
            out.push((prefix.to_string(), display));
        }
    }
}

/// Resolve a dot-notation path to a JSON value reference.
fn resolve_dot_path<'a>(value: &'a serde_json::Value, path: &str) -> Option<&'a serde_json::Value> {
    let mut current = value;
    for segment in path.split('.') {
        current = current.get(segment)?;
    }
    Some(current)
}

/// Check if a key name refers to a sensitive value that should be redacted.
fn is_sensitive_key(key: &str) -> bool {
    let lower = key.to_lowercase();
    lower.contains("api_key") || lower.contains("token") || lower.contains("secret")
}
```
